from django.db import models
from django.contrib.auth import get_user_model
from django.core.exceptions import ValidationError

User = get_user_model()


class Tag(models.Model):
    user = models.ForeignKey(
        User,
        on_delete=models.CASCADE,
        related_name='tags',
    )
    name = models.CharField(max_length=50, verbose_name='Nome')
    created_at = models.DateTimeField(auto_now_add=True, verbose_name='Criada em')

    class Meta:
        constraints = [
            models.UniqueConstraint(
                fields=['user', 'name'],
                name='unique_tag_per_user',
            )
        ]
        ordering = ['name']
        verbose_name = 'Tag'
        verbose_name_plural = 'Tags'
        indexes = [
            models.Index(fields=['user', 'name']),
        ]

    def __str__(self):
        return self.name

    def clean(self):
        super().clean()
        if self.name:
            self.name = self.name.strip().lower()
        if not self.name:
            raise ValidationError({'name': 'O nome da tag não pode estar vazio.'})

    def save(self, *args, **kwargs):
        self.full_clean()
        super().save(*args, **kwargs)
```

# tags/forms.py
from django import forms
from django.core.exceptions import ValidationError
from .models import Tag


class TagForm(forms.ModelForm):
    class Meta:
        model = Tag
        fields = ['name']
        widgets = {
            'name': forms.TextInput(attrs={
                'class': 'form-input',
                'placeholder': 'Nome da tag',
                'maxlength': '50',
                'enterkeyhint': 'done',
                'autocapitalize': 'none',
            }),
        }
        labels = {'name': 'Nome da Tag'}

    def __init__(self, *args, user=None, **kwargs):
        self.user = user
        super().__init__(*args, **kwargs)

    def clean_name(self):
        name = self.cleaned_data.get('name', '').strip().lower()
        if not name:
            raise ValidationError('O nome da tag não pode estar vazio.')

        existing = Tag.objects.filter(user=self.user, name=name)
        if self.instance and self.instance.pk:
            existing = existing.exclude(pk=self.instance.pk)
        if existing.exists():
            raise ValidationError(f'Já existe uma tag com o nome "{name}".')

        return name

    def save(self, commit=True):
        tag = super().save(commit=False)
        if self.user and not tag.user_id:
            tag.user = self.user
        if commit:
            tag.save()
        return tag
```

from django.urls import reverse


class TagViewTest(TestCase):

            username='viewuser', password='testpass123'
        )

def save(self, commit=True):
    transaction = super().save(commit=False)

    if not self.user:
        raise ValidationError("User is required to create transaction")

    if not hasattr(transaction, 'user') or not transaction.user:
        transaction.user = self.user

    if not transaction.user_id:
        transaction.user_id = self.user.id

    if commit:

<div class="mb-4" x-data="tagInput()" x-init="init()">

    <div class="relative">
        <input type="text"
               x-model="query"
               @input.debounce.200ms="search()"
               @keydown.enter.prevent="addFirst()"
               class="form-input w-full"
               placeholder="Digitar tag..."
               autocomplete="off">

        {# Dropdown suggestions #}
        <ul x-show="suggestions.length > 0" @click.outside="suggestions = []"
            class="absolute z-10 w-full mt-1 bg-dark-700 border border-gray-600 rounded-md shadow-lg max-h-40 overflow-y-auto">
            <template x-for="s in suggestions" :key="s.id">
                <li @click="add(s)"
                    class="px-3 py-2 text-sm text-gray-200 hover:bg-dark-600 cursor-pointer"
                    x-text="s.name"></li>
            </template>
            <li x-show="query.trim() && !exactMatch"
                @click="createNew()"
                class="px-3 py-2 text-sm text-sky-400 hover:bg-dark-600 cursor-pointer">
                + Criar "<span x-text="query.trim()"></span>"
            </li>
        </ul>
    </div>

    {# Hidden select for form submission #}
    <select name="tags" id="id_tags" multiple class="hidden" x-ref="hiddenSelect">
        <template x-for="tag in selected" :key="tag.id">
            <option :value="tag.id" selected x-text="tag.name"></option>
        </template>
    </select>
</div>

<script>
function tagInput() {
    return {
        query: '',
        selected: [],
        suggestions: [],
        allTags: [],
        exactMatch: false,

        init() {
            // Load existing tags from the rendered select options
            const select = this.$refs.hiddenSelect || document.getElementById('id_tags');
            const existing = document.querySelectorAll('#id_tags option[selected]');
            existing.forEach(opt => {
                this.selected.push({ id: parseInt(opt.value), name: opt.textContent.trim() });
            });

            // Fetch all user tags for autocomplete
            fetch('/api/v1/tags/', { headers: { 'X-Requested-With': 'XMLHttpRequest' } })
                .then(r => r.json())
                .then(data => { this.allTags = data; })
                .catch(() => {});
        },

        search() {
            const q = this.query.trim().toLowerCase();
            if (!q) { this.suggestions = []; return; }
            const selectedIds = new Set(this.selected.map(t => t.id));
            this.suggestions = this.allTags.filter(
                t => t.name.includes(q) && !selectedIds.has(t.id)
            ).slice(0, 8);
            this.exactMatch = this.allTags.some(t => t.name === q);
        },

        add(tag) {
            if (!this.selected.find(t => t.id === tag.id)) {
                this.selected.push(tag);
            }
            this.query = '';
            this.suggestions = [];
            this.syncHidden();
        },

        addFirst() {
            if (this.suggestions.length > 0) {
                this.add(this.suggestions[0]);
            } else if (this.query.trim() && !this.exactMatch) {
                this.createNew();
            }
        },

        remove(tag) {
            this.selected = this.selected.filter(t => t.id !== tag.id);
            this.syncHidden();
        },

        createNew() {
            const name = this.query.trim().toLowerCase();
            if (!name) return;
            fetch('/api/v1/tags/', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-CSRFToken': document.querySelector('[name=csrfmiddlewaretoken]').value,
                    'X-Requested-With': 'XMLHttpRequest',
                },
                body: JSON.stringify({ name }),
            })
            .then(r => r.json())
            .then(tag => {
                this.allTags.push(tag);
                this.add(tag);
            })
            .catch(() => {});
        },

        syncHidden() {
            // Update hidden select for form submission
            const select = this.$refs.hiddenSelect;
            select.innerHTML = '';
            this.selected.forEach(tag => {
                const opt = document.createElement('option');
                opt.value = tag.id;
                opt.textContent = tag.name;
                opt.selected = true;
                select.appendChild(opt);
            });
        },
    };
}
</script>
